Prozedual

  • von oben nach unten
  • in überschaubare Teile zerlegt (funktionen)
let thisYear = 2018;
let birthFrog = 2016;
let birthMe = 1981;

let result = (thisYear - birthFrog) + (thisYear - birthMe);
//  => 39

let thisYear = 2018;
let birthFrog = 2016;
let birthMe = 1981;

let myAge = function() {
  return thisYear - birthMe;
}

let frogAge = function() {
  return thisYear - birthFrog;
}

let result = frogAge() + myAge();
//  => 39

  return thisYear - birthFrog;

  return (year || thisYear) - birthFrog;
}

let birthFrog = 2016;
let birthMe = 1981;

let thisYear = function(year) {
  let d = new Date();
  return d.getFullYear();
}

let myAge = function(year) {
  return (year || thisYear()) - birthMe;
}

let frogAge = function(year) {
  return (year || thisYear()) - birthFrog;
}

let result = frogAge() + myAge();
//  => 39

let resultNexYear = frogAge(2019) + myAge(2019);
//  => 41

Arbeit ohne Callback

  result.push(a[i] + 10);

Arbeit mit Callback

let a = [1, 2, 3];
let result = [];

a.forEach(function(item){
  result.push(item + 10)
})

result; // => [11, 12, 13]
